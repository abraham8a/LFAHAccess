// ---- LOCAL PERSISTENT COUNTER (NO API) ---- const STORAGE_KEY = "lfah_valid_scans_2026"; let validScans = 0; // Load saved counter function loadCounter() { const stored = localStorage.getItem(STORAGE_KEY); validScans = stored ? parseInt(stored, 10) || 0 : 0; countDisplay.innerText = validScans; } // Save counter function saveCounter() { localStorage.setItem(STORAGE_KEY, String(validScans)); } // Reset counter with PIN document.getElementById('resetScanBtn').addEventListener('click', () => { const pin = prompt("Enter PIN to reset counter:"); if (pin === config.RESET_PIN) { validScans = 0; saveCounter(); countDisplay.innerText = validScans; alert("Counter reset on this device."); } else if (pin !== null) { alert("Incorrect PIN."); } }); // ---- SCAN HANDLER ---- function onScanSuccess(decodedText) { if (isProcessing) return; isProcessing = true; let isValid = false; // Exact match rule if (decodedText.trim() === config.VALID_PARAMETER) { isValid = true; } triggerUI(isValid); } // ---- UI + COUNTER UPDATE ---- async function triggerUI(isValid) { overlay.style.display = 'flex'; if (isValid) { validScans++; saveCounter(); countDisplay.innerText = validScans; overlay.className = 'valid-bg'; overlayText.innerText = 'VALID'; playSound(1200, 'sine'); } else { overlay.className = 'flash-red'; overlayText.innerText = 'INVALID'; playSound(120, 'square'); } setTimeout(() => { overlay.style.display = 'none'; overlay.className = ''; isProcessing = false; }, 2500); } // ---- INIT ---- loadCounter();
