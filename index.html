// CONFIGURATION const APP_CONFIG = { VALID_PARAMETER: "AsambleaRegional2026", INVALID_PARAMETER: "", RESET_PIN: "1234", GLOBAL_NAMESPACE: "lfah-access-global-event-2026" }; // GLOBAL STATE let validScans = 0; let currentSessionId = 0; let isProcessing = false; let audioCtx; const setupDiv = document.getElementById('setup'); const mainUIDiv = document.getElementById('mainUI'); const overlay = document.getElementById('overlay'); const countDisplay = document.getElementById('count'); const overlayText = document.getElementById('overlayText'); const config = APP_CONFIG; // CountAPI base const COUNT_API_BASE = `https://api.countapi.xyz`; // Fetch global state async function fetchGlobalState() { try { // Fetch session ID let sessionRes = await fetch(`${COUNT_API_BASE}/get/${config.GLOBAL_NAMESPACE}/session_id`); let sessionData = await sessionRes.json(); currentSessionId = sessionData.value || 0; } catch { currentSessionId = 0; } try { // Fetch scans for this session let scansRes = await fetch(`${COUNT_API_BASE}/get/${config.GLOBAL_NAMESPACE}/scans_session_${currentSessionId}`); let scansData = await scansRes.json(); validScans = scansData.value || 0; } catch { validScans = 0; } countDisplay.innerText = validScans; } // Sync every 3 seconds setInterval(fetchGlobalState, 3000); // Start button document.getElementById('startBtn').addEventListener('click', async () => { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); setupDiv.style.display = 'none'; mainUIDiv.style.display = 'flex'; countDisplay.innerText = "..."; await fetchGlobalState(); startScanner(); }); // Reset button document.getElementById('resetScanBtn').addEventListener('click', async () => { const pin = prompt('Enter PIN to reset global counter:'); if (pin !== config.RESET_PIN) { if (pin !== null) alert('Incorrect PIN.'); return; } try { // Increment session ID = resets counter await fetch(`${COUNT_API_BASE}/hit/${config.GLOBAL_NAMESPACE}/session_id`); await fetchGlobalState(); alert('Global counter reset successfully!'); } catch { alert('Error connecting to sync server.'); } }); // Sound function playSound(freq, waveType) { if (!audioCtx) return; const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.type = waveType || 'sine'; osc.frequency.setValueAtTime(freq, audioCtx.currentTime); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 0.5); osc.stop(audioCtx.currentTime + 0.5); } // Date logic function checkDateLogic(decodedText) { const today = new Date().toISOString().split('T')[0]; const parts = decodedText.trim().split('|'); if (parts.length === 2) { return today >= parts[0] && today <= parts[1]; } if (parts.length === 1) { return decodedText.trim() === today; } return false; } // Scan handler function onScanSuccess(decodedText) { if (isProcessing) return; isProcessing = true; let isValid = false; let explicit = false; if (config.VALID_PARAMETER && decodedText.trim() === config.VALID_PARAMETER) { isValid = true; explicit = true; } if (!isValid && config.INVALID_PARAMETER && decodedText.trim() === config.INVALID_PARAMETER) { isValid = false; explicit = true; } if (!explicit) { isValid = checkDateLogic(decodedText); } triggerUI(isValid); } // UI + global sync async function triggerUI(isValid) { overlay.style.display = 'flex'; if (isValid) { validScans++; countDisplay.innerText = validScans; overlay.className = 'valid-bg'; overlayText.innerText = 'VALID'; playSound(1200, 'sine'); // Push to global counter try { await fetch(`${COUNT_API_BASE}/hit/${config.GLOBAL_NAMESPACE}/scans_session_${currentSessionId}`); } catch (e) { console.error("Sync failed", e); } } else { overlay.className = 'flash-red'; overlayText.innerText = 'INVALID'; playSound(120, 'square'); } setTimeout(() => { overlay.style.display = 'none'; overlay.className = ''; isProcessing = false; }, 2500); } // Scanner function startScanner() { const html5QrCode = new Html5Qrcode("reader"); const renderConfig = { fps: 15, qrbox: (vw, vh) => { let size = Math.floor(Math.min(vw, vh) * 0.7); return { width: size, height: size }; } }; html5QrCode.start( { facingMode: "environment" }, renderConfig, onScanSuccess ).catch(err => alert("Camera Error: " + err)); }
